# Create DB
``` SQL
-- 1. 유저 정보
CREATE TABLE USERS (
    USER_ID         VARCHAR2(50) PRIMARY KEY,
    PASSWORD        VARCHAR2(100) NOT NULL,
    NICKNAME        VARCHAR2(50) UNIQUE NOT NULL,
    EMAIL           VARCHAR2(100),
    CREATED_AT      DATE DEFAULT SYSDATE,
    TOTAL_TOKENS    NUMBER DEFAULT 0
);

-- 2. 퀘스트
CREATE TABLE QUESTS (
    QUEST_ID        NUMBER PRIMARY KEY,
    TITLE           VARCHAR2(200) NOT NULL,
    DESCRIPTION     CLOB,
    REWARD_TOKENS   NUMBER NOT NULL,
    LOCATION        VARCHAR2(255), -- 위도 경도 값을 넣어 파싱해서 쓰기
    CREATOR_ID      VARCHAR2(50) REFERENCES USERS(USER_ID),
    STATUS          VARCHAR2(20) DEFAULT 'OPEN', -- OPEN, TAKEN, DONE, CANCELLED
    CREATED_AT      DATE DEFAULT SYSDATE,
    DEADLINE        DATE
);

-- 3. 퀘스트 수주
CREATE TABLE QUEST_APPLICATIONS (
    APPLICATION_ID  NUMBER PRIMARY KEY,
    QUEST_ID        NUMBER REFERENCES QUESTS(QUEST_ID),
    APPLICANT_ID    VARCHAR2(50) REFERENCES USERS(USER_ID),
    STATUS          VARCHAR2(20) DEFAULT 'PENDING', -- PENDING, APPROVED, REJECTED
    APPLIED_AT      DATE DEFAULT SYSDATE
);

-- 4. 퀘스트 히스토리
CREATE TABLE QUEST_HISTORY (
    HISTORY_ID      NUMBER PRIMARY KEY,
    QUEST_ID        NUMBER REFERENCES QUESTS(QUEST_ID),
    USER_ID         VARCHAR2(50) REFERENCES USERS(USER_ID),
    RESULT          VARCHAR2(20), -- SUCCESS, FAIL
    COMPLETED_AT    DATE
);

-- 5. 토큰 구매 및 사용
CREATE TABLE PAYMENTS (
    PAYMENT_ID      NUMBER PRIMARY KEY,
    USER_ID         VARCHAR2(50) REFERENCES USERS(USER_ID),
    AMOUNT          NUMBER NOT NULL,
    TYPE            VARCHAR2(20) NOT NULL, -- PURCHASE, QUEST_REWARD, REFUND
    DESCRIPTION     VARCHAR2(255),
    CREATED_AT      DATE DEFAULT SYSDATE
);

-- 6. 토큰 환급 신청
CREATE TABLE WITHDRAWALS (
    WITHDRAWAL_ID   NUMBER PRIMARY KEY,
    USER_ID         VARCHAR2(50) REFERENCES USERS(USER_ID),
    AMOUNT          NUMBER NOT NULL,
    STATUS          VARCHAR2(20) DEFAULT 'REQUESTED', -- REQUESTED, APPROVED, REJECTED
    REQUESTED_AT    DATE DEFAULT SYSDATE
);

```
